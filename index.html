Git workflow

Short work


git checkout -b private_feature_branch
touch file1.txt
git add file1.txt
git commit -am "WIP"
						

git checkout master
git merge --squash private_feature_branch
git commit -v
						

git push
git branch -d private_feature_branch

Long work


git checkout -b private_feature_branch
touch file1.txt
git add file1.txt
git commit -am "WIP"
...
git commit -am "WIP"
..
git commit -am "WIP"
						

git rebase --interactive master
